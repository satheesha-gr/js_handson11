Q7. Write a function that accepts parameter n and returns factorial of n

        function fact(n) {

            if (n < 0) {
              return "Error: Factorial of a negative number does not exist.";
            }
        
            else if (n === 0 || n === 1) {
              return 1;
            }
        
            else {
              return n * fact(n - 1);
            }
          }
          console.log(fact(5));

Output

ans7
